Compliance Posture *_policy_assignments.json, *_secure_score.json Policy compliance, secure scores, findings

Sample Integration Code

Example Python script to parse JSON and create CSV for Excel import:

import json import pandas as pd import glob # Example: Parse role assignments into CSV all_assignments = [] for file in glob.glob("out/*_role_assignments.json"): with open(file, 'r') as f: data = json.load(f) for assignment in data: all_assignments.append({ 'Subscription': file.split('_')[0], 'PrincipalName': assignment.get('principalName'), 'Role': assignment.get('roleDefinitionName'), 'Scope': assignment.get('scope'), 'PrincipalType': assignment.get('principalType') }) # Create DataFrame and export to CSV df = pd.DataFrame(all_assignments) df.to_csv('rbac_assignments_for_excel.csv', index=False) print(f"Exported {len(df)} role assignments to CSV")
